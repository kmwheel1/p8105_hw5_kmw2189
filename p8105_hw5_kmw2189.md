replace_iris = function(col_name){
  

  if(is.numeric(col_name))
    {
    new_value = replace_na(col_name, mean(col_name, na.rm = TRUE))
  } 
  
  else if(is.character(col_name))
    {
    new_value = replace_na(col_name, "virginica")
  }
  
  new_value
}
